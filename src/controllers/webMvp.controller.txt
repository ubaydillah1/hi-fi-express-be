import { Request, Response } from "express";
import { query } from "../db/connection";

const DEMO_ACCOUNT_ID = 1;

const goalEnumMap: Record<string, string> = {
  "Get my first job in tech": "GET_FIRST_JOB",
  "Switch to a developer role": "SWITCH_DEVELOPER_ROLE",
  "Improve my coding skills": "IMPROVE_CODING_SKILLS",
  "Prepare for technical interviews": "PREPARE_INTERVIEWS",
  "Build a strong portfolio": "BUILD_PORTFOLIO",
  "Understand market demands": "UNDERSTAND_MARKET",
};

const toNumber = (value: unknown, fallback = 0) => {
  const parsed = Number(value);
  return Number.isFinite(parsed) ? parsed : fallback;
};

async function ensureDemoAccount() {
  await query(
    `
    INSERT IGNORE INTO user_accounts (id, email, name)
    VALUES (?, ?, ?)
    `,
    [DEMO_ACCOUNT_ID, "demo@wirapath.com", "Alex Rahman"]
  );
}

export class WebMvpController {
  testDb = async (_req: Request, res: Response): Promise<void> => {
    try {
      const rows = await query<any[]>("SELECT * FROM user_accounts LIMIT 5");

      res.status(200).json({
        success: true,
        message: "Backend Express connected to WiraPath MySQL database",
        data: rows,
      });
    } catch (error: any) {
      res.status(500).json({
        success: false,
        message: "Database connection failed",
        error: error.message || String(error),
      });
    }
  };

  getOnboarding = async (_req: Request, res: Response): Promise<void> => {
    try {
      await ensureDemoAccount();

      const profileRows = await query<any[]>(
        `
        SELECT
          ua.id,
          ua.name,
          ua.email,
          p.first_name AS firstName,
          p.last_name AS lastName,
          p.university,
          p.field_of_study AS fieldOfStudy,
          p.graduation_year AS graduationYear
        FROM user_accounts ua
        LEFT JOIN profiles p ON p.user_id = ua.id
        WHERE ua.id = ?
        LIMIT 1
        `,
        [DEMO_ACCOUNT_ID]
      );

      const goalRows = await query<any[]>(
        `
        SELECT g.name
        FROM user_goals ug
        JOIN goals g ON g.id = ug.goal_id
        WHERE ug.user_id = ?
        ORDER BY g.id ASC
        `,
        [DEMO_ACCOUNT_ID]
      );

      const documentRows = await query<any[]>(
        `
        SELECT
          document_type AS type,
          file_name AS fileName,
          file_url AS fileUrl
        FROM user_documents
        WHERE user_id = ?
        ORDER BY id ASC
        `,
        [DEMO_ACCOUNT_ID]
      );

      res.status(200).json({
        success: true,
        profile: profileRows[0] || null,
        goals: goalRows.map((item) => item.name),
        documents: documentRows,
      });
    } catch (error: any) {
      res.status(500).json({
        success: false,
        message: "Failed to fetch onboarding data",
        error: error.message || String(error),
      });
    }
  };

  saveOnboarding = async (req: Request, res: Response): Promise<void> => {
    const {
      firstName,
      lastName,
      university,
      fieldOfStudy,
      graduationYear,
      goals = [],
      cvFileName,
      transcriptFileName,
      githubConnected = false,
    } = req.body;

    try {
      await ensureDemoAccount();

      const fullName = `${firstName || ""} ${lastName || ""}`.trim() || "Demo User";

      await query(`UPDATE user_accounts SET name = ? WHERE id = ?`, [
        fullName,
        DEMO_ACCOUNT_ID,
      ]);

      await query(
        `
        INSERT INTO profiles
          (user_id, first_name, last_name, university, field_of_study, graduation_year)
        VALUES (?, ?, ?, ?, ?, ?)
        ON DUPLICATE KEY UPDATE
          first_name = VALUES(first_name),
          last_name = VALUES(last_name),
          university = VALUES(university),
          field_of_study = VALUES(field_of_study),
          graduation_year = VALUES(graduation_year)
        `,
        [
          DEMO_ACCOUNT_ID,
          firstName || null,
          lastName || null,
          university || null,
          fieldOfStudy || null,
          graduationYear || null,
        ]
      );

      await query(`DELETE FROM user_goals WHERE user_id = ?`, [DEMO_ACCOUNT_ID]);

      for (const goal of goals as string[]) {
        await query(`INSERT IGNORE INTO goals (name) VALUES (?)`, [goal]);

        const goalRows = await query<any[]>(
          `SELECT id FROM goals WHERE name = ? LIMIT 1`,
          [goal]
        );

        if (goalRows.length > 0) {
          await query(
            `INSERT IGNORE INTO user_goals (user_id, goal_id) VALUES (?, ?)`,
            [DEMO_ACCOUNT_ID, goalRows[0].id]
          );
        }
      }

      if (cvFileName) {
        await query(
          `
          INSERT INTO user_documents (user_id, document_type, file_name, file_url)
          VALUES (?, 'cv', ?, ?)
          ON DUPLICATE KEY UPDATE
            file_name = VALUES(file_name),
            file_url = VALUES(file_url),
            uploaded_at = CURRENT_TIMESTAMP
          `,
          [DEMO_ACCOUNT_ID, cvFileName, `/uploads/${cvFileName}`]
        );
      }

      if (transcriptFileName) {
        await query(
          `
          INSERT INTO user_documents (user_id, document_type, file_name, file_url)
          VALUES (?, 'transcript', ?, ?)
          ON DUPLICATE KEY UPDATE
            file_name = VALUES(file_name),
            file_url = VALUES(file_url),
            uploaded_at = CURRENT_TIMESTAMP
          `,
          [DEMO_ACCOUNT_ID, transcriptFileName, `/uploads/${transcriptFileName}`]
        );
      }

      // Sinkron ringan ke tabel users utama milik backend teman.
      // Ini aman untuk demo: jika email demo tidak ada di users, query ini tidak mengubah apa pun.
      const firstGoal = Array.isArray(goals) && goals.length > 0 ? goals[0] : null;
      const mappedGoal = firstGoal ? goalEnumMap[firstGoal] || null : null;
      await query(
        `
        UPDATE users
        SET
          first_name = ?,
          last_name = ?,
          university = ?,
          field_of_study = ?,
          graduation_year = ?,
          achievement_goal = COALESCE(?, achievement_goal),
          cv_url = COALESCE(?, cv_url),
          transcript_url = COALESCE(?, transcript_url),
          onboarding_completed = 1
        WHERE email = ?
        `,
        [
          firstName || null,
          lastName || null,
          university || null,
          fieldOfStudy || null,
          graduationYear ? Number(graduationYear) : null,
          mappedGoal,
          cvFileName ? `/uploads/${cvFileName}` : null,
          transcriptFileName ? `/uploads/${transcriptFileName}` : null,
          "demo@wirapath.com",
        ]
      );

      res.status(200).json({
        success: true,
        message: "Onboarding data saved successfully",
        githubConnected,
      });
    } catch (error: any) {
      res.status(500).json({
        success: false,
        message: "Failed to save onboarding data",
        error: error.message || String(error),
      });
    }
  };

  getDashboardSummary = async (_req: Request, res: Response): Promise<void> => {
    try {
      await ensureDemoAccount();

      const userRows = await query<any[]>(
        `
        SELECT
          ua.name,
          p.first_name AS firstName
        FROM user_accounts ua
        LEFT JOIN profiles p ON p.user_id = ua.id
        WHERE ua.id = ?
        LIMIT 1
        `,
        [DEMO_ACCOUNT_ID]
      );

      const skillRows = await query<any[]>(
        `
        SELECT
          COUNT(DISTINCT skill_name) AS skillsMapped,
          ROUND(AVG(current_score)) AS readinessScore,
          SUM(CASE WHEN priority = 'Critical' THEN 1 ELSE 0 END) AS criticalGaps,
          SUM(CASE WHEN current_score >= 70 THEN 1 ELSE 0 END) AS strengths
        FROM (
          SELECT skill_name, MAX(current_score) AS current_score, MAX(required_score) AS required_score, MAX(priority) AS priority
          FROM user_skills
          WHERE user_id = ?
          GROUP BY skill_name
        ) AS deduped_skills
        `,
        [DEMO_ACCOUNT_ID]
      );

      const user = userRows[0];
      const skills = skillRows[0];

      res.status(200).json({
        name: user?.firstName || user?.name || "User",
        streak: 3,
        readinessScore: toNumber(skills?.readinessScore),
        trend: "+8%",
        skillsMapped: toNumber(skills?.skillsMapped),
        criticalGaps: toNumber(skills?.criticalGaps),
        strengths: toNumber(skills?.strengths),
      });
    } catch (error: any) {
      res.status(500).json({
        success: false,
        message: "Failed to fetch dashboard summary",
        error: error.message || String(error),
      });
    }
  };

  getSkillGap = async (_req: Request, res: Response): Promise<void> => {
    try {
      const rows = await query<any[]>(
        `
        SELECT
          skill_name AS skill,
          MAX(current_score) AS current,
          MAX(required_score) AS required,
          MAX(demand) AS demand,
          MAX(priority) AS priority
        FROM user_skills
        WHERE user_id = ?
        GROUP BY skill_name
        ORDER BY (MAX(required_score) - MAX(current_score)) DESC
        `,
        [DEMO_ACCOUNT_ID]
      );

      res.status(200).json(rows);
    } catch (error: any) {
      res.status(500).json({
        success: false,
        message: "Failed to fetch skill gap",
        error: error.message || String(error),
      });
    }
  };

  getMarketDemand = async (_req: Request, res: Response): Promise<void> => {
    try {
      const rows = await query<any[]>(
        `
        SELECT
          rank_order AS rank,
          skill_name AS skill,
          CONCAT(ROUND(MAX(jobs_count) / 1000, 1), 'k jobs') AS jobs,
          MAX(trend_score) AS trend,
          MAX(bar_width) AS barWidth
        FROM market_demands
        GROUP BY rank_order, skill_name
        ORDER BY rank_order ASC
        `
      );

      res.status(200).json(rows);
    } catch (error: any) {
      res.status(500).json({
        success: false,
        message: "Failed to fetch market demand",
        error: error.message || String(error),
      });
    }
  };
}
